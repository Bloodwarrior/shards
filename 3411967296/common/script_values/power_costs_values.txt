		add = {

			value = 200
			format = "BASE_VALUE_FORMAT"
		}

		if = {
			limit = {
				scope:severly_injure = yes
			}
			add = {
				desc = "severly_injure_option"
				value = 150
				# format = "BASE_VALUE_FORMAT"
			}
		}
		if = {
			limit = {
				scope:smite = yes
			}
			add = {

			}
		}

		multiply = {

			value = dominion_multiplier
		}

	}
}
