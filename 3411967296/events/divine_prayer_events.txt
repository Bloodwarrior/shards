divine_prayer.9000 = {
	hidden = yes

	immediate = {
		save_scope_as = divinity
		scope:divinity.faith = {
			random_faith_ruler = {
				limit = {
					is_a_divinity = no
					highest_held_title_tier > 1
					is_adult = yes
					is_landed = yes
				}
				weight = {
					base = 0
					modifier = {
						add = { value = ai_greed divide = 10}
						always = yes
					}
					modifier = {
						add = { value = ai_zeal divide = 10}
						always = yes
					}
					modifier = {
						add = 25
						has_trait = ambitious
					}
					# Has a need
					modifier = {
						add = 40
						has_need = yes
					}

					# Has a grave need
					modifier = {
						add = 60
						has_grave_need = yes
					}

					modifier = {
						factor = { value = highest_held_title_tier multiply = highest_held_title_tier }
						always = yes
					}
				}
				save_scope_as = petitioner
				if = {
					limit = {
						scope:divinity = {
							has_character_flag = refusing_prayers
						}
					}
					if = {
						limit = { scope:petitioner.faith = root.faith }
						check_for_heresy = yes
					}
					add_opinion = {
						target = scope:divinity
						modifier = refused_boon
					}
					
				}
				else = {
					choose_prayer_effect = yes
				}
				
			}
		}
	}
}

divine_prayer.9001 = {
	hidden = yes

	immediate = {
		save_scope_as = divinity
		random_ruler = {
			limit = {
				is_a_divinity = no
				highest_held_title_tier > 1
				is_adult = yes
				NOT = {
					has_faith = scope:divinity.faith
				}
				is_landed = yes
			}
			weight = {
				base = 0

				modifier = {
					add = { value = ai_greed divide = 5}
					always = yes
				}
				modifier = {
					add = { value = ai_zeal divide = -10}
					always = yes
				}
				modifier = {
					add = 25
					has_trait = ambitious
				}
				# Has a need
				modifier = {
					add = 40
					has_need = yes
				}

				# Has a grave need
				modifier = {
					add = 60
					has_grave_need = yes
				}
				modifier = {
					factor = 5
					highest_held_title_tier = 4
				}
				modifier = {
					factor = 6
					highest_held_title_tier = 3
				}
				modifier = {
					factor = 0.5
					faith = {
						faith_hostility_level = {
							target = scope:divinity.faith
							value >= faith_hostile_level
						}
					}
					
				}
				modifier = {
					factor = 0.1
					faith = {
						faith_hostility_level = {
							target = scope:divinity.faith
							value >= faith_evil_level
						}
					}
					
				}
				
			}
			save_scope_as = petitioner
				if = {
					limit = {
						scope:divinity = {
							has_character_flag = refusing_prayers
						}
					}
					if = {
						limit = { scope:petitioner.faith = root.faith }

					}
					add_opinion = {
						target = scope:divinity
						modifier = refused_boon
					}
					
				}
				else = {
					choose_prayer_effect = yes
				}
		}
		
	}
}


# Petiotioner asking for blessing of war
divine_prayer.0001 = {
	title = divine_prayer.0001.title
	desc = divine_prayer.0001.desc
	theme = faith

	left_portrait = root
	
	right_portrait = scope:petitioner

	immediate = {
		save_scope_as = divinity
		save_scope_as = actor
		scope:petitioner = {
			save_scope_as = recipient
		}
	}
	
	# minor boon
	option = {
		name = divine_prayer.0001.a
		trigger = {
			piety >= bless_cost
		}
		scope:divinity = {
			add_piety =  {value = bless_cost multiply = -1}
		}
		scope:petitioner = {
			add_blessing = { BLESSING = war }
		}
		show_as_unavailable = {always = yes}
		gave_minor_boon = yes
	}

	# Medium boon
	option = {
		name = divine_prayer.0001.b
		trigger = {
			piety >= {value = bless_cost multiply = 3}
		}
		show_as_unavailable = {always = yes}
		scope:divinity = {
			add_piety =  {value = bless_cost multiply = -3}
		}
		scope:petitioner = {
			custom_tooltip = blessing_trice_tooltip
			custom_description = {
				text = increase_investment_text
				object = this
				value = 6
				add_blessing = { BLESSING = war }
				add_blessing = { BLESSING = war }
				add_blessing = { BLESSING = war }
			}
		}
		gave_medium_boon = yes
	}

	# # Major boon
	# option = {
	# 	name = divine_prayer.0001.c
	# 	gave_major_boon = yes
	# }
	
	option = {
		name = divine_prayer.0001.d
        ai_chance = { base = 0 }
		scope:petitioner = {
			if = {
				limit = { scope:petitioner.faith = root.faith }
				check_for_heresy = yes
			}
			add_opinion = {
				target = scope:divinity
				modifier = refused_boon
			}			
		}
		
	}
}
